The errors stem from a fundamental misunderstanding of Verilog-2001 `case` statement syntax. The `case` statement requires a colon (`:`) after the `case` keyword and before each case value. The testbench also had the same issue. The `default` case must also be properly formatted.

```verilog
module alu_ops (
  input [7:0] a,
  input [7:0] b,
  input [2:0] opcode,
  output reg [7:0] result,
  output reg carry_out,
  output reg zero
);

  always @(*) begin
    case (opcode)
      3'b000: // ADD
        result = a + b;
        carry_out = (a + b) >> 7;
      3'b001: // SUB
        result = a - b;
        carry_out = (a < b);
      3'b010: // AND
        result = a & b;
        carry_out = 1'b0;
      3'b011: // OR
        result = a | b;
        carry_out = 1'b0;
      3'b100: // XOR
        result = a ^ b;
        carry_out = 1'b0;
      default: // Default case for unexpected opcode
        result = 8'b0;
        carry_out = 1'b0;
    endcase

    zero = (result == 8'b0);
  end

endmodule
```
